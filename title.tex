			\vfill

			{ \small \textbf{\@ttitle} \par}
			\vskip 1cm
			\vfill
			\begin{flushleft}
				\begin{tabular}[t]{rl}
					Author: &\@author\\
					E-Mail: & \@email\\
					Date: & \@submitDate
				\end{tabular}
			\end{flushleft}
		\end{center}
		\restoregeometry
	\end{titlepage}
}
